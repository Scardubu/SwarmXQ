# ═══════════════════════════════════════════════════════════════════════════════
# SwarmX Python Runtime — Production Dockerfile
#
# Multi-stage build for minimal final image size.
#
# Production hardening:
#   [DKPY-FIX-01] HEALTHCHECK unified to /health (matches server.py [SRV-FIX-02]
#                 and docker-compose.yml healthcheck directive).
#   [DKPY-FIX-02] Data dir created + chowned in single RUN after useradd.
#   [DKPY-FIX-03] Removed duplicate `pip install -e ".[all]"` in the app stage.
#                 The deps stage already installs all extras. The app stage only
#                 re-installs the editable package itself (no extras), saving 4–8
#                 minutes on every full rebuild.
#   [DKPY-ENH-01] PYTHONDONTWRITEBYTECODE + PYTHONUNBUFFERED for prod discipline.
#   [DKPY-ENH-02] Consistent --no-cache-dir across all pip invocations.
#   [DKPY-ENH-03] SWARM_ROOT exported for vertical config loader.
#   [DKPY-ENH-04] tini as PID 1 — SIGTERM correctly forwarded to Python child.
#   [DKPY-ENH-05] Python base image pinned to a specific patch release.
#
# Build:  docker build -f Dockerfile.python -t swarmx-python .
# Run:    docker run --rm -p 8787:8787 swarmx-python
# ═══════════════════════════════════════════════════════════════════════════════

ARG PYTHON_VERSION=3.12.9
FROM python:${PYTHON_VERSION}-slim-bookworm AS base

# [DKPY-ENH-01] Set before every RUN so all layers inherit these settings.
# PYTHONDONTWRITEBYTECODE — skip .pyc generation (saves ~30 MB in large installs)
# PYTHONUNBUFFERED       — stdout/stderr always flushed; critical for log aggregators
ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1

# System dependencies:
#   tini          — lightweight init (PID 1) for correct signal forwarding [DKPY-ENH-04]
#   curl          — healthcheck probe + doctor command
#   git           — needed by pip packages that build from source at install time
#   build-essential, libffi-dev, libsqlite3-dev — native extension build dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
        build-essential \
        curl \
        git \
        libffi-dev \
        libsqlite3-dev \
        tini \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# ── Dependency layer (cached unless pyproject.toml or requirements.txt change) ─
# Separated from the application layer so a code change does NOT bust the pip cache.
FROM base AS deps

COPY pyproject.toml requirements.txt ./

# [DKPY-ENH-02] --no-cache-dir throughout to keep each layer as lean as possible.
# Install in this order:
#   1. Upgrade pip/wheel/setuptools (avoids wheel build failures on older pip)
#   2. requirements.txt (heavy ML/vector packages — benefits most from cache)
#   3. swarmx package with all optional framework adapters (once, in deps stage)
RUN pip install --no-cache-dir --upgrade pip wheel setuptools \
    && pip install --no-cache-dir -r requirements.txt \
    && pip install --no-cache-dir -e ".[all]"

# ── Application layer ──────────────────────────────────────────────────────────
# Copies only the source directories actually needed at runtime.
# .dockerignore excludes __pycache__, *.pyc, .git, node_modules, env.local, etc.
FROM deps AS app

COPY src/        ./src/
COPY agents/     ./agents/
COPY skills/     ./skills/
COPY workflows/  ./workflows/
COPY configs/    ./configs/
COPY templates/  ./templates/
COPY brain/      ./brain/

# [DKPY-FIX-03] Re-install the editable package WITHOUT extras — all extras
# (framework adapters, ML libs) are already installed in the deps stage above.
# Re-installing with [all] would redundantly re-download and re-build every
# optional adapter (~400 MB) on every code change, with no benefit.
RUN pip install --no-cache-dir -e "."

# ── Runtime user (non-root) ────────────────────────────────────────────────────
# [DKPY-FIX-02] Create user FIRST, then create + chown the data dir in a single
# RUN so the swarmx user owns it from initial container start (not as root).
RUN useradd --create-home --shell /bin/bash --uid 1001 swarmx \
    && mkdir -p /data/swarmx \
    && chown -R swarmx:swarmx /data/swarmx

USER swarmx

# ── Runtime environment ────────────────────────────────────────────────────────
ENV SWARM_HOME=/data/swarmx \
    # [DKPY-ENH-03] Bundle root so the vertical config loader and routing.yaml
    # loader can resolve configs/verticals/*.yaml relative to /app.
    SWARM_ROOT=/app

EXPOSE 8787

# [DKPY-FIX-01] /health is the canonical production route.
# Matches server.py [SRV-FIX-02] and docker-compose.yml healthcheck directive.
HEALTHCHECK --interval=10s --timeout=5s --start-period=30s --retries=6 \
    CMD curl -sf http://localhost:8787/health || exit 1

# [DKPY-ENH-04] tini as PID 1 ensures SIGTERM is correctly delivered to the
# Python process during `docker stop`. Without tini, SIGTERM goes to PID 1
# (bash) which does NOT forward it, leaving the server to be SIGKILL'd after
# the stop timeout — silently dropping in-flight requests.
ENTRYPOINT ["/usr/bin/tini", "--"]
CMD ["python", "-m", "swarmx", "serve", "--host", "0.0.0.0", "--port", "8787"]
